Script Source Code
#!/bin/bash

# Ask for project name
read -p "Enter your Rust project name: " project_name

# Validate project name
if [ -z "$project_name" ]; then
    echo "Project name cannot be empty."
    exit 1
fi

# Create new Rust project
cargo new "$project_name"
cd "$project_name"

# Create directory structure
mkdir -p src/{api,auth,config,db/{migrations,models},handlers,utils}
mkdir -p tests/{integration,unit}
mkdir -p benches docs

# Create mod.rs files
touch src/api/mod.rs
touch src/auth/mod.rs
touch src/config/mod.rs
touch src/db/mod.rs
touch src/handlers/mod.rs
touch src/utils/mod.rs

# Update Cargo.toml with dependencies
cat > Cargo.toml << EOF
[package]
name = "$project_name"
version = "0.1.0"
edition = "2021"

[dependencies]
tokio = { version = "1.0", features = ["full"] }
axum = "0.6"
serde = { version = "1.0", features = ["derive"] }
serde_json = "1.0"
sqlx = { version = "0.7", features = ["runtime-tokio-rustls", "postgres"] }
tracing = "0.1"
tracing-subscriber = "0.3"
dotenv = "0.15"
tower = "0.4"
tower-http = { version = "0.4", features = ["trace"] }

[dev-dependencies]
tokio-test = "0.4"
EOF

# Create main.rs
cat > src/main.rs << 'EOF'
mod api;
mod auth;
mod config;
mod db;
mod handlers;
mod utils;

#[tokio::main]
async fn main() {
    println!("Hello, Rust!");
}
EOF

# Create .env
cat > .env << EOF
DATABASE_URL=postgres://user:password@localhost:5432/dbname
EOF

# Create docker-compose.yml
cat > docker-compose.yml << EOF
version: '3.8'
services:
  db:
    image: postgres:latest
    environment:
      POSTGRES_USER: user
      POSTGRES_PASSWORD: password
      POSTGRES_DB: dbname
    ports:
      - "5432:5432"
EOF

echo "Rust project '$project_name' created successfully!"
echo "Run 'cargo build' to compile the project"
Usage
  1. Save the script to a file (e.g., create_rust_project.sh)
  2. Make it executable: chmod +x create_rust_project.sh
  3. Run: ./create_rust_project.sh
  4. Follow the prompts to set up your project
